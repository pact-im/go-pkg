// ErrorType succeeds if err is not nil and is of the expected type.
// New code should use [ErrorIs] instead.
//
// Expected can be one of:
//
//	func(error) bool
//
// Function should return true if the error is the expected type.
//
//	type struct{}, type &struct{}
//
// A struct or a pointer to a struct.
// Fails if the error is not of the same type as expected.
//
//	type &interface{}
//
// A pointer to an interface type.
// Fails if err does not implement the interface.
//
//	reflect.Type
//
// Fails if err does not implement the [reflect.Type].
func ( error,  interface{}) Comparison {
	return func() Result {
		switch expectedType := .(type) {
		case func(error) bool:
			return cmpErrorTypeFunc(, )
		case reflect.Type:
			if .Kind() == reflect.Interface {
				return cmpErrorTypeImplementsType(, )
			}
			return cmpErrorTypeEqualType(, )
		case nil:
			return ResultFailure("invalid type for expected: nil")
		}

		 := reflect.TypeOf()
		switch {
		case .Kind() == reflect.Struct, isPtrToStruct():
			return cmpErrorTypeEqualType(, )
		case isPtrToInterface():
			return cmpErrorTypeImplementsType(, .Elem())
		}
		return ResultFailure(fmt.Sprintf("invalid type for expected: %T", ))
	}
}
