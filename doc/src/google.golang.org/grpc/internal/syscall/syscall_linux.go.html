// GetCPUTime returns the how much CPU time has passed since the start of this process.
func () int64 {
	var  unix.Timespec
	if  := unix.ClockGettime(unix.CLOCK_PROCESS_CPUTIME_ID, &);  != nil {
		logger.Fatal()
	}
	return .Nano()
}
