// scanner provides a bufio.Scanner the ability to Unscan,
// which allows the current token to be read again
// after the next Scan.
type scanner struct {
	*bufio.Scanner

	unscanned bool
}

func ( io.Reader) *scanner {
	return &scanner{Scanner: bufio.NewScanner()}
}

func ( *scanner) () bool {
	if .unscanned {
		.unscanned = false
		return true
	}
	return .Scanner.Scan()
}

// Unscan stops the scanner from advancing its position
// for the next Scan.
//
// Bytes and Text will return the same token after next Scan
// that they do right now.
func ( *scanner) () {
	.unscanned = true
}
