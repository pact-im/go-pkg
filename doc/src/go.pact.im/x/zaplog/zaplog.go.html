			LevelKey:       "level",
			TimeKey:        "time",

			CallerKey:      "caller",

			EncodeTime:     zapcore.ISO8601TimeEncoder,
			EncodeDuration: zapcore.StringDurationEncoder,
			EncodeCaller:   zapcore.FullCallerEncoder,
		}),
		zapcore.AddSync(),
		zap.DebugLevel,
	), zap.WithCaller(true))
}
