// Per RFC 8017, Section 9.1
//
//     EM = MGF1 xor DB || H( 8*0x00 || mHash || salt ) || 0xbc
//
// where
//
//     DB = PS || 0x01 || salt
//
// and PS can be empty so
//
//     emLen = dbLen + hLen + 1 = psLen + sLen + hLen + 2
//

// incCounter increments a four byte, big-endian counter.
func ( *[4]byte) {
	if [3]++; [3] != 0 {
		return
	}
	if [2]++; [2] != 0 {
		return
	}
	if [1]++; [1] != 0 {
		return
	}
	[0]++
}
