// timestamp is an unprocessed timestamp.
type timestamp uint64

// batch represents a batch of trace events.
// It is unparsed except for its header.
type batch struct {
	m    threadID
	time timestamp
	data []byte
}

// threadID is the runtime-internal M structure's ID. This is unique
// for each OS thread.
type threadID int64
